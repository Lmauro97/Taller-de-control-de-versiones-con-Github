Proceso punto6
	Definir Contactos como texto;
	Dimension Contactos[3,3];
	Definir C como entero;
	Definir F como entero;
	Para C<-0 hasta 2 Hacer
		Para F<-0 Hasta 2 Hacer
			Contactos[F,C]<-"";
		FinPara
	FinPara
	Definir Opcion como entero;
	Definir Opcion2 como entero;
	Definir NumeroTelefonico como texto;
	Definir Nombre como texto;
	Definir NombreEmpresa como texto;
	Definir R como logico; //comprueba si se repite
	Repetir
		Escribir "Contactos";
		Escribir "1. Agregar Contactos";
		Escribir "2. Buscar Contacto";
		Escribir "3. Borrar Contacto";
		Escribir "4. Cerrar programa";		
		Leer Opcion;
		Segun Opcion hacer 
			1:
				Limpiar Pantalla;
				Escribir "Cual es el telefono que desea agregar: ";
				Leer NumeroTelefonico;
				R<-falso;
				para F <- 0 Hasta 2 Hacer
					Si NumeroTelefonico=Contactos[F,0]       
						Entonces
						R<-Verdadero;
					FinSi
				FinPara
				Si R=Falso Entonces 
					Para F<-0 hasta 2 Hacer
						Si Contactos[F,0]="" Entonces
							Contactos[F,0]<-NumeroTelefonico;
							Escribir "Nombre: ";
							Leer Nombre;
							Contactos[F,1]<-Nombre;
							Escribir "Organización: ";
							Leer NombreEmpresa;
							Contactos[F,2]<-NombreEmpresa;
							F<-2;
							R<-Verdadero;
							Limpiar pantalla;
						FinSi
					FinPara
					
					Si R=Falso Entonces
						Escribir "Por favor borre un numero para poder guardar el nuevo";
						Esperar 5 Segundos;
					FinSi
				SiNo
					R<-Falso;
					Limpiar Pantalla;
					Escribir "Contacto ya esta registrado, por favor vuelva a seleccionar que desea hacer";
					Esperar 5 Segundos;
					Limpiar Pantalla;
				FinSi
			2:
				Limpiar Pantalla;
				Escribir "Buscar Contacto";
				Leer Nombre;
				R<-falso;
				Para F<-0 hasta 2 Hacer
					Si Contactos[F,1]=Nombre Entonces
						Escribir "Nombre de contacto: ",Contactos[F,1];
						Escribir "Número de documento: ",Contactos[F,0];
						Escribir "Organización: ",Contactos[F,2];
						Esperar 10 Segundos;
						R<-verdadero;
					FinSi
				FinPara
				Si R=falso Entonces
					Escribir "Datos no encontrados";
					Escribir "Contactos actuales:";
					Escribir "------------------------------";
					Para F<-0 hasta 2 Hacer
						Escribir"";
						Para C<-0 Hasta 2 Hacer
							Escribir Sin Saltar Contactos[F,C];
							Escribir Sin Saltar " - ";
						FinPara
					FinPara
					Esperar 10 Segundos;
				Finsi
				Limpiar Pantalla;
			3:
				Limpiar Pantalla;
				Escribir "Borrar contacto: ";
				Escribir"-------------------";
				Para F<-0 hasta 2 Hacer
					Escribir"";
					Escribir Sin Saltar F+1," - ";
					Para C<-0 Hasta 2 Hacer
						Escribir Sin Saltar Contactos[F,C]," - ";
					FinPara
				FinPara
				Escribir "";
				Escribir "Que numero de contacto desea borrar";
				Leer Opcion2;
				Segun Opcion2 Hacer
					1:
						Para C<-0 Hasta 2 Hacer
							Contactos[0,C]<-"";
						FinPara
					2:
						Para C<-0 Hasta 2 Hacer
							Contactos[1,C]<-"";
						FinPara
					3:
						Para C<-0 Hasta 2 Hacer
							Contactos[2,C]<-"";
						FinPara
					De Otro Modo:
						Escribir "Opcion no valida";
						Esperar 5 segundos;
				FinSegun
				Limpiar Pantalla;
		FinSegun       
	Hasta que Opcion=4;
FinProceso
