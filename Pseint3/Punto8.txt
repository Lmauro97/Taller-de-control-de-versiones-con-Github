Proceso Punto8
	Definir Nombre, Usuario, Licencia, Fecha, Estado, Grupo Como Caracter;
	Definir C,F,Opcion como entero;
	Dimension Grupo[8,5];
	Para C<-0 hasta 4 Hacer
		Para F<-0 Hasta 7 Hacer
			Grupo[F,C]<-"";
		FinPara
	FinPara
	Repetir
		Escribir "El Maestro";
		Escribir "--------------------";
		Escribir "1-Registrar al curso";
		Escribir "2-Consultar usuarios";
		Escribir "3-Resultados";
		Escribir "4-Cerrar programa";
		Leer Opcion;
		Segun Opcion Hacer
			1:
				Limpiar Pantalla;
				Escribir "Registrar usuario";
				Escribir "--------------------------";
				Escribir "Nombre: ";
				Leer Nombre;
				Para F<-0 Hasta 7 Hacer
					Si Grupo[F,0]="" Entonces
						Escribir "Documento: ";
						Leer Usuario;
						Escribir "Licencia a obtener: ";
						Leer Licencia;
						Escribir "Fecha: ";
						Leer Fecha;
						Grupo[F,0]<-Usuario;
						Grupo[F,1]<-Nombre;
						Grupo[F,2]<-Licencia;
						Grupo[F,3]<-Fecha;
						Grupo[F,4]<-"Sin examen";
						Escribir "Usuario creado";
						F<-7;
						Esperar Tecla;
						Limpiar Pantalla;
					FinSi
				FinPara
			2:
				Limpiar Pantalla;
				Escribir "Consultar usuario";
				Escribir "____________________________";
				Escribir "Número de documento:";
				Leer Usuario;
				Para F<-0 Hasta 7 Hacer
					Si Grupo[F,0]=Usuario Entonces
						Para C<-0 Hasta 4 Hacer
							Escribir Grupo[F,C], " - ";
						FinPara
						F<-7;
						Esperar Tecla;
						Usuario<-"";
					FinSi
				FinPara
				Si Usuario<>"" Entonces
					Escribir "Persona no esta registrada en el curso";
					Esperar Tecla;
				FinSi
				Limpiar Pantalla;
			3:
				Limpiar Pantalla;
				Escribir "Resultados";
				Escribir "_______________________________________";
				Escribir "Número de documento:";
				Leer Usuario;
				Para F<-0 Hasta 7 Hacer
					Si Grupo[F,0]=Usuario Entonces
						Para C<-0 Hasta 4 Hacer
							Escribir Sin Saltar Grupo[F,C]," - ";
						FinPara
						Escribir "";
						Escribir "Usuario aprueba curso: ";
						Escribir "1-Si";
						Escribir "2-No";
						Leer Opcion;
						Segun Opcion Hacer
							1:
								Grupo[F,4]<-"Aprobado";
							2: 
								Grupo[F,4]<-"No aprobado";
							De Otro Modo:
								Escribir "No valido";
						FinSegun
						F<-7;
						Usuario<-"";
					FinSi
				FinPara
				Si Usuario<>"" Entonces
					Escribir "Persona no esta registrada en el curso";
				FinSi
				Escribir "";
				Esperar Tecla;
				Limpiar Pantalla;
			De Otro Modo:
				Escribir "Opcion no valida";
		FinSegun
	Hasta Que Opcion=4
FinProceso
