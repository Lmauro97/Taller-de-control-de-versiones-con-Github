Proceso punto7
	Definir Marca, Placa, NombreCompleto, NumeroTelefono, Contactos Como Caracter;
	Definir C,F,Opcion como entero;
	Dimension Contactos[5,4];
	Para C<-0 hasta 3 Hacer
		Para F<-0 Hasta 4 Hacer
			Contactos[F,C]<-"";
		FinPara
	FinPara
	Repetir
		Escribir "El guardián";
		Escribir "--------------------";
		Escribir "1-Ingresar al parqueadero";
		Escribir "2-Retirar del parqueadero";
		Escribir "3-Consultar vehiculo";
		Escribir "4-Cerrar programa";
		Leer Opcion;
		Segun Opcion Hacer
			1:
				Limpiar Pantalla;
				Escribir "Ingreso al parqueadero";
				Escribir "--------------------------";
				Escribir "Número de placa: ";
				Leer Placa;
				Para F<-0 Hasta 4 Hacer
					Si Contactos[F,0]="" Entonces
						Escribir "Marca: ";
						Leer Marca;
						Escribir "Nombre completo del dueño";
						Leer NombreCompleto;
						Escribir "Número completo";
						Leer NumeroTelefono;
						Contactos[F,0]<-Placa;
						Contactos[F,1]<-Marca;
						Contactos[F,2]<-NombreCompleto;
						Contactos[F,3]<-NumeroTelefono;
						Escribir "vehiculo cargado";
						F<-4;
						Esperar Tecla;
						Limpiar Pantalla;
					FinSi
				FinPara
			2:
				Limpiar Pantalla;
				Escribir "Retirar del parqueadero";
				Escribir "____________________________";
				Escribir "Número de placa:";
				Leer Placa;
				Para F<-0 Hasta 4 Hacer
					Si Contactos[F,0]=Placa Entonces
						Contactos[F,0]<-"";
						Contactos[F,1]<-"";
						Contactos[F,2]<-"";
						Contactos[F,3]<-"";
						Escribir "Vehiculo Retirado";
						Placa<-"";
						F<-4;
						Esperar Tecla;
					FinSi
				FinPara
				Si Placa<>"" Entonces
					Escribir "Vehiculo no esta registrado en el inventario";
					Esperar Tecla;
				FinSi
				Limpiar Pantalla;
			3:
				Limpiar Pantalla;
				Escribir "Consultar vehiculo en el parqueadero";
				Escribir "_______________________________________";
				Escribir "Número de placa:";
				Leer Placa;
				Para F<-0 Hasta 4 Hacer
					Si Contactos[F,0]=Placa Entonces
						Placa<-"";
						Para C<-0 Hasta 3 Hacer
							Escribir Sin Saltar Contactos[F,C]," - ";
						FinPara
					FinSi
				FinPara
				Si Placa<>"" Entonces
					Escribir "Vehiculo no esta en el parqueadero";
					Escribir "Vehiculos en el parqueadero";
					Para F<-0 Hasta 4 Hacer
						Si Contactos[F,0]<>"" Entonces
							Escribir Sin Saltar F+1," - ";
							Para C<-0 Hasta 3 Hacer
								Escribir Sin Saltar Contactos[F,C]," - ";
							FinPara
						FinSi
					FinPara
				FinSi
				Escribir "";
				Esperar Tecla;
				Limpiar Pantalla;
			De Otro Modo:
				Escribir "Opcion no valida";
		FinSegun
	Hasta Que Opcion=4
FinProceso
