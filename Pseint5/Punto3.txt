SubProceso Punto1() 
	Definir vector Como Entero;
	Dimension vector[5];
	Definir i Como Entero;
	Definir numero Como Entero;
	para i<-0 Hasta 4 Hacer
		Escribir "Por favor ingrese numero para la posición: ",i;
		Leer numero;
		vector[i]<-numero;
	FinPara
	Limpiar Pantalla;
	Para i<-0 Hasta 4 Hacer
		Escribir "[",i,"]"," = ",vector[i];
	FinPara
	Esperar Tecla;
FinSubProceso

SubProceso Punto2()

	Dimension vector[20];
	Definir i Como Entero;
	Definir numero Como Entero;
	
	para i<-0 Hasta 19 Hacer
		Escribir "Por favor ingrese numero para la posición: ",i;
		Leer numero;
		vector[i]<-numero;
	FinPara
	
	Limpiar Pantalla;
	Escribir sin saltar "Números pares: ";
	i<-0;
	Repetir
		Si vector[i]%2=0 Entonces
			Escribir Sin Saltar vector[i],", ";
		FinSi
		i<-i+1;
	Hasta Que i=20
	i<-0;
	Escribir "";
	Escribir Sin Saltar "Números impares: ";
	Repetir
		Si vector[i]%2=1 Entonces
			Escribir Sin Saltar vector[i],", ";
		FinSi
		i<-i+1;
	Hasta Que i=20
	Esperar Tecla;
FinSubProceso

SubProceso Punto3
	Definir i Como Entero;
	Definir j Como Entero;
	Definir n Como Entero;
	Para i<-2 Hasta 1000 Hacer
		n<-0;
		//contador de divisores que tendra el numero a evaluar
		Para j<-1 hasta i Hacer
			si i%j = 0 Entonces
				n<-n+1;
				Si n>2 Entonces
					j<-i;
				FinSi
				//detiene el calculo para ahorrar tiempo de procesamiento
			FinSi
		FinPara
		Si n=2 Entonces
			Escribir i;
		FinSi
	FinPara
	Esperar Tecla;
FinSubProceso

SubProceso Punto4()

	n[0,0]<-"01";n[0,1]<-"02";n[0,2]<-"03";n[0,3]<-"04";n[0,4]<-"05";
	n[1,0]<-"06";n[1,1]<-"07";n[1,2]<-"08";n[1,3]<-"09";n[1,4]<-"10";
	n[2,0]<-"11";n[2,1]<-"12";n[2,2]<-"13";n[2,3]<-"14";n[2,4]<-"15";
	n[3,0]<-"16";n[3,1]<-"17";n[3,2]<-"18";n[3,3]<-"19";n[3,4]<-"20";
	Para i<-0 Hasta 3 Hacer
		Escribir "";
		Si i%2=0 Entonces
			Para j<-0 Hasta 4 Hacer
				Escribir Sin Saltar n[i,j]," ";
			FinPara
		SiNo
			Para j<-4 Hasta 4 Hacer
				Escribir Sin Saltar n[i,j]," ";
				j<-j-2;
				Si j=-2 Entonces
					j<-4;
				FinSi
			FinPara
		FinSi
	FinPara
	Escribir "";
	Esperar Tecla;
FinSubProceso

SubProceso Punto5()
	Definir valorx, valory,i,xy Como Entero;
	Dimension xy[10,10];
	Para valory<-0 Hasta 9 Hacer
		Para valorx<-0 Hasta 9 Hacer
			xy[valory,valorx]<-(valory+1)*(valorx+1);
		FinPara
	FinPara
	Escribir "                            Columnas ";
	Escribir Sin Saltar"     ";
	Para valorx<-0 Hasta 9 Hacer
		Escribir Sin Saltar"  ",valorx,"  ";
	FinPara
	Escribir "";
	Para valory<-1 Hasta 10 Hacer
		Para valorx<-1 Hasta 10 Hacer
			Si valorx=1 Entonces
				Segun valory Hacer
					3:Escribir Sin Saltar"F ";
					4:Escribir Sin Saltar"i ";
					5:Escribir Sin Saltar"l ";	
					6:Escribir Sin Saltar"a ";	
					7:Escribir Sin Saltar"s ";	
				FinSegun
				Si valory<=2 o valory>=8 Entonces
					Escribir Sin Saltar"  ";
				FinSi
				Escribir Sin Saltar valory-1," ";
			FinSi
			Si valory=10 Entonces
				Escribir Sin Saltar" ";
			SiNo
				Escribir Sin Saltar"  ";
			FinSi
			Escribir Sin Saltar valorx,"x",valory;
		FinPara
		Escribir "";
	FinPara
	Escribir "Posicion de operacion que se desea tener";
	Escribir "Fila:";
	Leer valory;
	Escribir "Columna:";
	Leer valorx;
	Escribir "La multiplicacion es igual a: ",xy[valory,valorx];
	Esperar Tecla;
FinSubProceso
